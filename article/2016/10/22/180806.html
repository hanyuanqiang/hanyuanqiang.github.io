字符串工具类：StringUtil.java判断字符串是否为空

2016-10-22 18:08:06 原创 分类：Util类汇总

代码如下：

package com.hyq.util;

public class StringUtil {

	public static boolean isEmpty(String str){
		if("".equals(str)|| str==null){
			return true;
		}else{
			return false;
		}
	}
	
	public static boolean isNotEmpty(String str){
		return !isEmpty(str);
	}
	
	public static void main(String[] args) {
		String string1 = "";
		System.out.println("string1是否为空："+isEmpty(string1));
		String string2 = null;
		System.out.println("string2是否为空："+isEmpty(string2));
		String string3 = "随便";
		System.out.println("string3是否为空："+isEmpty(string3));
	}
	
}
