1、Spring Mvc拦截器

Spring Mvc中如果要实现自己的拦截器，有两种方式：一是继承HandlerInterceptor接口，并且实现该接口的三个方法，二是通过实现HandlerInterceptorAdapter，并且重写所需方法。下面举例：

首先定义两个拦截器：MyInterceptor.java和MyInterceptor2.java

package com.hyq.interceptor;

import org.springframework.web.servlet.HandlerInterceptor;
import org.springframework.web.servlet.ModelAndView;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Created by Administrator on 2016/11/11.
 */
public class MyInterceptor implements HandlerInterceptor{

    public boolean preHandle(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse, Object o) throws Exception {
        System.out.println("MyInterceptor的preHandle方法...");
        return true;
    }

    public void postHandle(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse, Object o, ModelAndView modelAndView) throws Exception {
        System.out.println("MyInterceptor的postHandle方法...");
    }

    public void afterCompletion(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse, Object o, Exception e) throws Exception {
        System.out.println("MyInterceptor的afterCompletion方法...");
    }
}

package com.hyq.interceptor;

import com.hyq.mvc.UserController;
import org.springframework.ui.ModelMap;
import org.springframework.web.servlet.HandlerInterceptor;
import org.springframework.web.servlet.ModelAndView;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Created by Administrator on 2016/11/11.
 */
public class MyInterceptor2 implements HandlerInterceptor{


    public boolean preHandle(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse, Object o) throws Exception {
        System.out.println("MyInterceptor2的preHandle方法...");
        return true;
    }

    public void postHandle(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse, Object o, ModelAndView modelAndView) throws Exception {
        System.out.println("MyInterceptor2的postHandle方法...");
    }

    public void afterCompletion(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse, Object o, Exception e) throws Exception {
        System.out.println("MyInterceptor2的afterCompletion方法...");
    }
}

然后在springMvc的配置文件中加入如下配置：

<mvc:interceptors>
        <!--拦截所有请求-->
        <bean class="com.hyq.interceptor.MyInterceptor2"></bean>
    </mvc:interceptors>
    <mvc:interceptors>
        <mvc:interceptor>
            <!--拦截指定请求路径的请求-->
            <mvc:mapping path="/user/*"/>
            <bean class="com.hyq.interceptor.MyInterceptor"></bean>
        </mvc:interceptor>
    </mvc:interceptors>

下面给出一个Controller类：UserController.java

package com.hyq.mvc;

import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.servlet.ModelAndView;

/**
 * Created by Administrator on 2016/11/11.
 */
@Controller
@RequestMapping(value = "/user")
public class UserController {

    @RequestMapping(value = "/add")

        System.out.println("添加用户操作...");

        modelAndView.addObject("name","张三");

    }

}

2、Spring Mvc 异常处理

  • 使用Spring Mvc提供的SimpleMappingExceptionResolver
该方法只需在spring mvc的配置文件中加入如下配置：
<!-- 总错误处理-->
    <bean id="exceptionResolver" class="org.springframework.web.servlet.handler.SimpleMappingExceptionResolver">
        <!--设置发生错误时的默认跳转页面,在jsp页面中可以使用${exception}来获取异常信息-->
        <property name="defaultErrorView">
            <value>/error.jsp</value>
        </property>
        <property name="defaultStatusCode">
            <value>500</value>
        </property>
        <!--下面属性表示在发生异常时给出警告-->
        <property name="warnLogCategory">
            <value>org.springframework.web.servlet.handler.SimpleMappingExceptionResolver</value>
        </property>
    </bean>

当Controller层发生异常时，SpringMvc会自动捕捉该异常，并且跳转到指定页面(此处是跳转到error.jsp页面)，并且可以在页面上通过EL表达式取得异常信息，即${exception}

  • 通过实现Spring的异常处理接口HandlerExceptionResolver来自定义自己的异常处理器，如：
package com.hyq.exception;

import org.springframework.web.servlet.HandlerExceptionResolver;
import org.springframework.web.servlet.ModelAndView;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Created by Administrator on 2016/11/11.
 */
public class MyException implements HandlerExceptionResolver {
    public ModelAndView resolveException(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse, Object o, Exception e) {
        System.out.println("下面输出异常信息....");
        e.printStackTrace();
        ModelAndView modelAndView = new ModelAndView();
        modelAndView.addObject("exception","这是异常信息(自定义异常处理类)");
        modelAndView.setViewName("/error.jsp");
        return modelAndView;
    }
}

在上面的UserController类中加find方法：

@RequestMapping(value = "/find")
    public ModelAndView find(Integer id) throws Exception {
        System.out.println("查找用户操作...");
        ModelAndView modelAndView = new ModelAndView();
        modelAndView.setViewName("/userList.jsp");
        if(id==null){
            throw new Exception();
        }
        return modelAndView;
    }

最后在Spring Mvc配置文件中加入如下配置：

<bean id="myException" class="com.hyq.exception.MyException"/>

浏览器地址栏中请求http://localhost:8080/user/find

同时也可在控制台中看到异常信息的输出。

3、Spring Mvc 上传图片

使用Spring Mvc进行文件上传时，需要在springmvc的配置文件中加入下面配置：

<bean id="multipartResolver"
        class="org.springframework.web.multipart.commons.CommonsMultipartResolver">
		<property name="defaultEncoding" value="UTF-8"/>  
	    <property name="maxUploadSize" value="10000000"/>
	</bean>

此外还要添加这两个jar包：com.springsource.org.apache.commons.fileupload-1.2.0.jar、com.springsource.org.apache.commons.io-1.4.0.jar

2、FileUploadController.java

package com.java1234.controller;

import java.io.File;

import javax.servlet.http.HttpServletRequest;

import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.multipart.MultipartFile;

@Controller
public class FileUploadController {

	@RequestMapping("/upload")
	public String uploadFile(@RequestParam("file") MultipartFile file,HttpServletRequest request) throws Exception{
		System.out.println("单文件上传");
		String filePath = request.getServletContext().getRealPath("/");
		System.out.println("原文件名：--"+file.getOriginalFilename()+"--");
		//可以根据下面语句判断上传的文件是否为空
		if(!"".equals(file.getOriginalFilename())){
			file.transferTo(new File(filePath+"upload/"+file.getOriginalFilename()));
		}else{
			System.out.println("未选择上传的文件");
		}
		System.out.println("------分割线------");
		return "redirect:success.html";
	}
	
	@RequestMapping("/upload2")
	public String uploadFiles(@RequestParam("file") MultipartFile[] files,HttpServletRequest request) throws Exception{
		System.out.println("多文件上传");
		String filePath = request.getServletContext().getRealPath("/");
		for(MultipartFile file:files){
			if("".equals(file.getOriginalFilename())){
				System.out.println("未选择上传的文件");
			}else{
				System.out.println("原文件名：--"+file.getOriginalFilename()+"--");
				file.transferTo(new File(filePath+"upload/"+file.getOriginalFilename()));
			}
		}
		System.out.println("------分割线------");
		return "redirect:success.html";
	}
	
}
