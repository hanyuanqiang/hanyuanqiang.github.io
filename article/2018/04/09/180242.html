1、自动发送弹幕

$("div[class='expression fl']").html("<button onclick='start()'>开始</button>");//替换表情
$("div").remove("div[class='unread-msgs fl']");	//移除站内信
$("div[class='fl noble--broadcast-box']").html("<button onclick='stop()'>暂停</button>");//替换喇叭
$("div[class='fl noble--barrage-box status--nobel-disabled']").html("<button onclick='end()'>结束</button>");//替换贵族弹幕
$("div[class='filter-keywords fl ']").html("<span style='color:red'>未开始</span>");//替换过滤
$("div").remove("div[class='hyperlink-chat fl']");//移除超级链接

var int;
function start(){
	$("div[class='filter-keywords fl ']").html("<span style='color:red'>正在执行</span>");//替换过滤
	var content = $(".cs-textarea").val();
	var intervalTime = 400;
	var tag = true;
	int = setInterval(function () {
		if (content == "") {
			$("div[class='filter-keywords fl ']").html("<span style='color:red'>请输入内容</span>");//替换过滤
			window.clearInterval(int);
		}else{
			if(tag){
		        $("#js-send-msg .cs-textarea").val(content);
		    }else{
		        $("#js-send-msg .cs-textarea").val(content+"~");
		    }
		    if ($("#js-send-msg .b-btn").text() == '发送') {
		    	$("#js-send-msg .b-btn").click();
		    	tag = !tag;
		    }
		}
	},intervalTime);
}
function stop(){
	$("div[class='filter-keywords fl ']").html("<span style='color:red'>暂停执行</span>");//替换过滤
	window.clearInterval(int);
	$(".cs-textarea").val(content);
}
function end(){
	$("div[class='filter-keywords fl ']").html("<span style='color:red'>结束执行</span>");//替换过滤
	window.clearInterval(int);
	$(".cs-textarea").val("");
}


2.统计发言人数

$("ul[class='r-else clearfix']").append("<li style='color:red'>发言人数统计：<span id='userNums'>"+0+"</span></li>");
$("ul[class='r-else clearfix']").append("<li style='color:red'><button onclick='start()'>开始</button></li>");
$("ul[class='r-else clearfix']").append("<li style='color:red'><button onclick='show()'>获取所有发言人</button></li>");
$("ul[class='r-else clearfix']").append("<li style='color:red'><button onclick='stop()'>结束统计</button></li>");
var users = new Array();
var index = 0;
var intervalTime = 1000 * 1;
var t4;
//统计发言的用户
function start(){
	t4 = setInterval(function () {
	    var textCont = $("li[class='jschartli hy-chat  ']");
	    textCont.each(function(){
	    	var userLevel = $(this).attr("data-level");
	      	var text = $(this).children(".text-cont").children(".name").children(".js-nick");
		    var nick = text.text().replace("：","");
		    var user = nick+"-"+userLevel;
	        if(users.indexOf(user)<0){
	            users[index] = user;
	            index++;
	        	$("#userNums").text(index);
	        }
	    });
	},intervalTime);
}

function show(){
	$("#js-stats-and-actions").html("<p style='background-color:white'>总人数是:"+users.length+","+users.join()+"</p>");
}

function stop(){
	window.clearInterval(t4);
}
