斗鱼自动发送弹幕脚本（粗糙版）

2018-04-10 02:02:42 原创 分类：生活、杂想 

$("div[class='expression fl']").html("<button onclick='start()'>开始</button>");//替换表情
$("div").remove("div[class='unread-msgs fl']");	//移除站内信
$("div[class='fl noble--broadcast-box']").html("<button onclick='stop()'>暂停</button>");//替换喇叭
$("div[class='fl noble--barrage-box status--nobel-disabled']").html("<button onclick='end()'>结束</button>");//替换贵族弹幕
$("div[class='filter-keywords fl ']").html("<span style='color:red'>未开始</span>");//替换过滤
$("div").remove("div[class='hyperlink-chat fl']");//移除超级链接

var int;
function start(){
	$("div[class='filter-keywords fl ']").html("<span style='color:red'>正在执行</span>");//替换过滤
	var content = $(".cs-textarea").val();
	var intervalTime = 400;
	var tag = true;
	int = setInterval(function () {
		if (content == "") {
			$("div[class='filter-keywords fl ']").html("<span style='color:red'>请输入内容</span>");//替换过滤
			window.clearInterval(int);
		}else{
			if(tag){
		        $("#js-send-msg .cs-textarea").val(content);
		    }else{
		        $("#js-send-msg .cs-textarea").val(content+"~");
		    }
		    if ($("#js-send-msg .b-btn").text() == '发送') {
		    	$("#js-send-msg .b-btn").click();
		    	tag = !tag;
		    }
		}
	},intervalTime);
}
function stop(){
	$("div[class='filter-keywords fl ']").html("<span style='color:red'>暂停执行</span>");//替换过滤
	window.clearInterval(int);
	$(".cs-textarea").val(content);
}
function end(){
	$("div[class='filter-keywords fl ']").html("<span style='color:red'>结束执行</span>");//替换过滤
	window.clearInterval(int);
	$(".cs-textarea").val("");
}
